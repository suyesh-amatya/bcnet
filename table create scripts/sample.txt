CREATE TABLE IF NOT EXISTS `bcnet`.`sample` (
  `uuid_` varchar(75) NOT NULL,
  `sampleDbId` bigint(20) NOT NULL AUTO_INCREMENT,
  `sampleCollectionId` varchar(255) DEFAULT NULL,
  `biobankId` varchar(255) NOT NULL,
  `hashedSampleId` varchar(255) NOT NULL,
  `hashedIndividualId` varchar(255) DEFAULT NULL,
  `materialType` varchar(255) NOT NULL,
  `container` varchar(45) DEFAULT NULL,
  `storageTemperature` varchar(45) DEFAULT NULL COMMENT 'Text string of letters with the temperature for the long-term storage of the biospecimens in the sample collection. Can be one or several of the following values:  Room temperature, +4 °C, -18 °C to -35 °C, -60 °C to -85 °C, Liquid nitrogen, Other. T',
  `sampledTime` datetime DEFAULT NULL,
  `anatomicalPartOntology` varchar(255) DEFAULT NULL COMMENT '(from lookup table)',
  `anatomicalPartOntologyVersion` varchar(255) DEFAULT NULL,
  `anatomicalPartOntologyCode` varchar(255) DEFAULT NULL,
  `anatomicalPartOntologyDescription` varchar(255) DEFAULT NULL COMMENT 'for the specified anatomicalPartOntology',
  `anatomicalPartFreeText` varchar(255) DEFAULT NULL,
  `sex` varchar(45) DEFAULT NULL,
  `ageLow` int(11) DEFAULT NULL,
  `ageHigh` int(11) DEFAULT NULL,  
  `ageUnit` varchar(45) DEFAULT NULL,
  `diseaseOntology` varchar(255) DEFAULT NULL COMMENT '(from lookup table)',
  `diseaseOntologyVersion` varchar(255) DEFAULT NULL,
  `diseaseOntologyCode` varchar(255) DEFAULT NULL,
  `diseaseOntologyDescription` varchar(255) DEFAULT NULL COMMENT 'for the specified diseaseOntology',
  `diseaseFreeText` varchar(255) DEFAULT NULL,
  `countryOfOrigin` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`sampleDbId`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
